import { useState } from 'react';

import ComponentLibrarySelector from '@/components/ComponentLibrarySelector';

import CodeEditor from '@/components/CodeEditor';
import ModifyControls from '@/components/ModifyControls';
import PublishButton from '@/components/PublishButton';
import { cn } from '@/lib/utils';
import { motion } from 'framer-motion';

// 模拟生成的代码数据
const mockGeneratedCode = `import React from 'react';
import { Button, Card, Input } from 'antd';

export default function GeneratedPage() {
  
  return (
    <div className="container mx-auto p-4">
      <Card title="Generated Page" bordered={false}>
        <h1 className="text-2xl font-bold mb-4">Welcome to My Page</h1>
        <Input placeholder="Enter text here" className="mb-4" />
        <Button type="primary">Submit</Button>
      </Card>
    </div>
  );
  
}`;

// 默认预览内容 - 简单的响应式页面示例

<div class="min-h-screen bg-gray-50 flex flex-col">
  <header class="bg-white shadow-sm">
    <div class="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-4">
      <h1 class="text-2xl font-bold text-gray-900">Sample Generated Page</h1> 
    </div>
  </header>
  
   <main class="flex-grow max-w-7xl w-full mx-auto px-4 sm:px-6 lg:px-8 py-8">
    <div class="bg-white shadow rounded-lg p-6">
      <h2 class="text-xl font-semibold mb-4">Welcome to Your Website</h2>
      <p class="text-gray-700 mb-6">This is a preview of your generated website. You can modify the content using natural language or by adjusting elements directly.</p>
      
      <div class="flex space-x-4">
        <button class="bg-blue-500 hover:bg-blue-600 text-white px-4 py-2 rounded-md">Primary Action</button>
        <button class="bg-gray-200 hover:bg-gray-300 text-gray-800 px-4 py-2 rounded-md">Secondary Action</button>
      </div>
    </div>
  </main>
  
  <footer class="bg-white border-t mt-auto">
    <div class="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-4">
      <p class="text-center text-gray-500 text-sm">Generated with AI Website Builder</p>
    </div>
  </footer>
</div>
`;

  const [componentLibrary, setComponentLibrary] = useState('antd');
  const [previewContent, setPreviewContent] = useState(defaultPreviewContent);
  const [generatedCode, setGeneratedCode] = useState(mockGeneratedCode);
  const [isGenerating, setIsGenerating] = useState(false);
  const [userPrompt, setUserPrompt] = useState('');

  // 模拟代码生成功能
  const handleGenerateCode = () => {
    if (!userPrompt.trim()) return;
    
    setIsGenerating(true);
    
    // 模拟API调用延迟
    setTimeout(() => {
      // 在实际应用中，这里会根据用户输入生成真实代码
      setIsGenerating(false);
      
      // 简单模拟根据用户输入修改预览内容
      const updatedPreview = defaultPreviewContent.replace(
        'Sample Generated Page', 
        userPrompt.length > 30 ? `${userPrompt.substring(0, 30)}...` : userPrompt
      );
      
      setPreviewContent(updatedPreview);
    }, 1500);
  };

  // 处理草图上传
  const handleSketchUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (file) {
      // 模拟处理草图文件
      setIsGenerating(true);
      
      setTimeout(() => {
        setIsGenerating(false);
        setPreviewContent(defaultPreviewContent.replace(
          'Sample Generated Page', 
          'From Sketch Input'
        ));
      }, 2000);
    }
  };

  // 处理修改请求
  const handleModifyRequest = (modification: string) => {
    if (!modification.trim()) return;
    
    setIsGenerating(true);
    
    setTimeout(() => {
      setIsGenerating(false);
      // 简单模拟修改效果
      const modifiedPreview = previewContent.replace(
        'Welcome to Your Website', 
        modification.length > 40 ? `${modification.substring(0, 40)}...` : modification

        <div className="container mx-auto px-4 py-4 flex justify-between items-center">

            <a href="#" className="text-gray-700 dark:text-gray-300 hover:text-blue-600 dark:hover:text-blue-400 font-medium transition-colors">
              首页
            </a>
            <a href="#" className="text-gray-700 dark:text-gray-300 hover:text-blue-600 dark:hover:text-blue-400 font-medium transition-colors">
              功能
            </a>
            <a href="#" className="text-gray-700 dark:text-gray-300 hover:text-blue-600 dark:hover:text-blue-400 font-medium transition-colors">
              教程
            </a>
            <a href="#" className="text-gray-700 dark:text-gray-300 hover:text-blue-600 dark:hover:text-blue-400 font-medium transition-colors">
              定价
            </a>
          </nav>
          
          <div className="flex items-center space-x-4">
            <button className="hidden md:flex items-center justify-center w-10 h-10 rounded-full bg-gray-100 dark:bg-gray-800 text-gray-700 dark:text-gray-300 hover:bg-gray-200 dark:hover:bg-gray-700 transition-colors">
              <i className="fa-solid fa-moon"></i>
            </button>
            <button className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded-lg font-medium transition-colors shadow-sm hover:shadow">
              登录
            </button>
          </div>
        </div>
      </header>

      <main className="container mx-auto px-4 py-8">
        {/* 英雄区域 */}
        <section className="mb-12 text-center max-w-3xl mx-auto">
          <h2 className="text-4xl md:text-5xl font-bold mb-4 bg-gradient-to-r from-gray-900 to-gray-700 dark:from-white dark:to-gray-300 bg-clip-text text-transparent">
            用自然语言或草图<br />生成专业网页代码
          </h2>
          <p className="text-lg text-gray-600 dark:text-gray-400 mb-8">
            无需编程经验，只需描述您的需求或上传草图，即可自动生成高质量、可直接部署的网页代码

                描述您的需求

              <ComponentLibrarySelector

                onSelect={setComponentLibrary}

                      "flex-1 py-2 px-4 text-sm font-medium transition-colors",

                        ? "bg-blue-600 text-white" 
                        : "bg-gray-100 dark:bg-gray-700 text-gray-700 dark:text-gray-300 hover:bg-gray-200 dark:hover:bg-gray-600"
                    )}
                    onClick={() => setInputMethod('sketch')}
                  >
                    <i className="fa-solid fa-paint-brush mr-1"></i> 草图上传
                  </button>
                </div>
                
                {inputMethod === 'text' ? (
                  <div>
                    <textarea
                      value={userPrompt}
                      onChange={(e) => setUserPrompt(e.target.value)}

                    ></textarea>
                    <div className="mt-4 flex justify-end">
                      <button
                        onClick={handleGenerateCode}
                        disabled={isGenerating || !userPrompt.trim()}
                        className={cn(
                          "px-6 py-2 rounded-lg font-medium transition-all shadow-sm hover:shadow",
                          (isGenerating || !userPrompt.trim())
                            ? "bg-gray-300 dark:bg-gray-600 text-gray-500 dark:text-gray-400 cursor-not-allowed"
                            : "bg-blue-600 hover:bg-blue-700 text-white"
                        )}
                      >
                        {isGenerating ? (
                          <>

                            生成中...

                          <>
                            <i className="fa-solid fa-code mr-2"></i>

                      accept="image/*"
                      onChange={handleSketchUpload}
                      className="hidden"

                    />

                      <div className="text-5xl text-gray-400 mb-3">
                        <i className="fa-solid fa-cloud-upload-alt"></i>
                      </div>
                      <p className="text-gray-600 dark:text-gray-400 mb-2">拖放草图图片到此处或</p>
                      <p className="text-blue-600 dark:text-blue-400 font-medium">点击上传</p>
                      <p className="text-xs text-gray-500 dark:text-gray-500 mt-2">支持 PNG, JPG 或 GIF (最大 10MB)</p>
                    </label>
                  </div>
                )}
              </div>
            </div>
            
            <ModifyControls onModify={handleModifyRequest} />
            
            <PublishButton />
          </div>
          
          {/* 右侧：预览和代码区域 */}
          <div className="lg:col-span-2 space-y-6">
            {/* 预览区域 */}
            <div className="bg-white dark:bg-gray-800 rounded-xl shadow-sm border border-gray-200 dark:border-gray-700 overflow-hidden transition-all hover:shadow-md">
              <div className="border-b border-gray-200 dark:border-gray-700 px-6 py-4 flex justify-between items-center">
                <h3 className="text-xl font-semibold flex items-center">
                  <i className="fa-solid fa-desktop text-blue-600 mr-2"></i>
                  实时预览
                </h3>
                <div className="flex space-x-2">
                  <button className="w-8 h-8 flex items-center justify-center rounded hover:bg-gray-100 dark:hover:bg-gray-700 transition-colors">
                    <i className="fa-solid fa-mobile-alt text-gray-600 dark:text-gray-400"></i>
                  </button>
                  <button className="w-8 h-8 flex items-center justify-center rounded hover:bg-gray-100 dark:hover:bg-gray-700 transition-colors">
                    <i className="fa-solid fa-tablet-alt text-gray-600 dark:text-gray-400"></i>
                  </button>
                  <button className="w-8 h-8 flex items-center justify-center rounded bg-gray-100 dark:bg-gray-700 transition-colors">
                    <i className="fa-solid fa-desktop text-gray-600 dark:text-gray-400"></i>
                  </button>
                </div>
              </div>
              
              <div className="relative h-[500px] overflow-auto bg-gray-50 dark:bg-gray-900">
                {isGenerating ? (
                  <div className="absolute inset-0 flex flex-col items-center justify-center bg-white/80 dark:bg-gray-800/80 backdrop-blur-sm z-10">
                    <div className="w-16 h-16 border-4 border-blue-200 border-t-blue-600 rounded-full animate-spin mb-4"></div>
                    <p className="text-lg font-medium text-gray-700 dark:text-gray-300">正在生成您的网页...</p>
                    <p className="text-gray-500 dark:text-gray-400 text-sm mt-1">
                      我们正在根据您的描述创建完美的网页
                    </p>
                  </div>
                ) : null}
                
                <div 
                  className="p-4 min-w-full min-h-full"
                  dangerouslySetInnerHTML={{ __html: previewContent }}
                />
                
                {/* 预览窗口控制栏模拟 */}
                <div className="absolute top-0 left-0 right-0 h-6 bg-gray-100 dark:bg-gray-800 flex items-center px-3 border-b border-gray-200 dark:border-gray-700">
                  <div className="flex space-x-2">
                    <div className="w-3 h-3 rounded-full bg-red-400"></div>
                    <div className="w-3 h-3 rounded-full bg-yellow-400"></div>
                    <div className="w-3 h-3 rounded-full bg-green-400"></div>
                  </div>
                  <div className="ml-4 text-xs text-gray-500 dark:text-gray-400">
                    preview.codegenius.com
                  </div>
                </div>
              </div>
            </div>
            
            {/* 代码区域 */}
            <CodeEditor code={generatedCode} />
          </div>
        </div>
      </main>

      <footer className="bg-white dark:bg-gray-800 border-t border-gray-200 dark:border-gray-700 mt-16">
        <div className="container mx-auto px-4 py-12">
          <div className="grid grid-cols-1 md:grid-cols-4 gap-8">
            <div>
              <div className="flex items-center space-x-2 mb-4">
                <div className="w-8 h-8 rounded-lg bg-blue-600 flex items-center justify-center">
                  <i className="fa-solid fa-code text-white"></i>
                </div>
                <h3 className="text-lg font-bold">CodeGenius</h3>
              </div>
              <p className="text-sm text-gray-600 dark:text-gray-400 mb-4">
                用AI技术赋能网页开发，让创意快速变为现实
              </p>
              <div className="flex space-x-4">
                <a href="#" className="text-gray-400 hover:text-blue-600 dark:hover:text-blue-400 transition-colors">
                  <i className="fa-brands fa-twitter"></i>
                </a>
                <a href="#" className="text-gray-400 hover:text-blue-600 dark:hover:text-blue-400 transition-colors">
                  <i className="fa-brands fa-github"></i>
                </a>
                <a href="#" className="text-gray-400 hover:text-blue-600 dark:hover:text-blue-400 transition-colors">
                  <i className="fa-brands fa-linkedin"></i>
                </a>
              </div>
            </div>
            
            <div>
              <h4 className="font-semibold mb-4">产品</h4>
              <ul className="space-y-2 text-sm">
                <li><a href="#" className="text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400">功能</a></li>
                <li><a href="#" className="text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400">定价</a></li>
                <li><a href="#" className="text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400">教程</a></li>
                <li><a href="#" className="text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400">更新日志</a></li>
              </ul>
            </div>
            
            <div>
              <h4 className="font-semibold mb-4">公司</h4>
              <ul className="space-y-2 text-sm">
                <li><a href="#" className="text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400">关于我们</a></li>
                <li><a href="#" className="text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400">联系方式</a></li>
                <li><a href="#" className="text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400">招贤纳士</a></li>
                <li><a href="#" className="text-gray-600 dark:text-gray-400 hover:text-blue-600 dark:hover:text-blue-400">媒体报道</a></li>

              </ul>
            </div>
          </div>
          
          <div className="border-t border-gray-200 dark:border-gray-700 mt-10 pt-6 flex flex-col md:flex-row justify-between items-center">
            <p className="text-sm text-gray-500 dark:text-gray-400">
